# Node는 각 링크드리스트의 원소를 만드는 클래스에요
# 메소드는 없고 필드만 있는 클래스!
class Node:
    def __init__(self, data, next):
        self.data = data # 현재 노드에 담길 데이터
        self.next = next # 링크로 연결될 노드를 담아요

# LinkedList는 말그대로.. Linked List
class LinkedList:
    def __init__(self):
        # 각 필드들을 초기화해줘요
        self.no = 0 # 현재 원소 갯수
        self.head = None # 첫 원소
        self.current = None # 현재 탐색중인 원소
    
    def __len__(self):
        return self.no
    
    def add_first(self, data):
        ptr = self.head # 곧 없어질 self.head를 담기 위한 임시 변수
        
        self.head = self.current = Node(data, ptr)
        # self.head "였던것"을 next로 가지는 새로운 노드를 생성!
        # 생성된 노드를 self.head, self.current에 대입해줘요

        self.no = self.no + 1 # 워소 갯수를 하나 늘리기!

    def add_last(self, data):
        # 맨 마지막에 노드를 추가하는 메소드!

        if self.head == None: # head가 None이다? 빈 리스트다? -> 처음에 원소 넣는거랑 동치!
            self.add_first(data) # 넣으라고 한 값을 리스트 처음에 추가

        else: # 빈 리스트가 아닐때!
            
            # 마지막 원소를 찾기 위한 부단한 노력
            ptr = self.head
            # 마지막 원소를 찾아서 저장할 변수.
            # 앞으로 나아갈거기 때문에 일단은 head로 초기화
            
            while ptr.next is not None: # ptr.next가 None이다? 다음원소가 없다? 이번 원소(ptr)가 끝이다!
                ptr = ptr.next # 아니면.. 다음원소로 이동
            
            ptr.next = self.current = Node(data, None)
            # 다음원소가 None인 (지정되지 않은) 새 노드를 만들어서 current와 "현재 원소의 다음"으로 지정한다.
            
            # 원소 갯수 하나 늘리기
            self.no = self.no + 1

    def remove_first(self):
        # 맨 처음 노드를 삭제하는 메소드!
        # 사실 이름은 삭제하는거긴 한데.. 그냥 링크를 끊어주면 파이썬 인터프리터에서 알아서 삭제할거기 때문에
        # 그냥 첫 노드의 링크를 끊는거라고 생각하면 돼요

        if self.head is not None: # 리스트가 비어있지 않다면
            self.head = self.current = self.head.next
            # 첫번째 원소의 다음 원소(두번째 원소)를 첫번째 원소와 currnet로 지정
        self.no = self.no - 1 # 원소 갯수 하나 줄이기
    
    def remove_last(self):
        if self.head is not None: # 리스트가 비어있지 않다면
            if self.head.next is None: # 리스트의 두번째 원소(처음의 다음)가 비어있다면 == 원소가 1개라면
                self.remove_first()
            else: # 원소가 여러개일 때
                # 마지막 노드를 찾기 위한 험난한 과정
                ptr = self.head # 현재 탐색중인 노드를 담을 변수
                pre = self.head # 현재 탐색중인 노드의 이전 노드를 담을 변수

                # 왜 위에 초기화가 저렇게 됐나요?
                #   ->  self.head와 self.head.next로 초기화해도 되는데
                #       원소가 하나인 리스트에서는 아래줄에서는 에러가 발생함!
                #       (But 이 코드상에서는 원소가 하나인 상황을 배제하여서 오류는 나지 않음)
                #       (So head랑 head.next로 초기화해도 문제 없음)

                # 현재의 다음원소가 없을때까지 반복하기!
                # -> 현재 원소가 마지막일 때 까지!
                while ptr.next is not None:
                    pre = ptr       # 현재 원소는 이전 원소가 됨
                    ptr = ptr.next  # 다음 원소는 현재 원소가 됨

                pre.next = None # 이전 원소의 다음 (마지막 원소)를 지정해제
                self.current = pre 
                self.no = no - 1 # 원소 갯수 하나 줄이기
    def remove(self, p):
        # 입력받은 원소를 지우는 메소드
        if self.head is not None: # 원소가 비어있지 않으면
            if p is self.head: # 입력받은게 첫번쨰 원소면..??
                self.remove_first()
            else:
                # p를 찾기 위한 험난한 노력
                ptr = self.head
                while ptr.next is not p: # ptr.next가 p가 아닐때 까지 반복? -> 결국엔 ptr.next가 p가 됨
                    ptr = ptr.next
                    if ptr is None: # ptr이 None이 되면..?? "끝까지 찾았지만.. 이 원소에는 p가 없다.. 간절함이 부족한 탓일까.."
                        return
                ptr.next = p.next
                # ptr.next가 p.next가 되면? p는 건너뛰게 됨.
                # a -> b -> c -> d인 리스트가 있다고 가정을 하자.
                # p가 c이면? ptr은 b가 됨.
                # 여기서 ptr.next를 p.next로 지정하게 되면
                # ptr.next이 d가 되고, 따라서
                # a -> b -> d가 됨. p인 c가 삭제되었다!!

                self.current = ptr
                self.no = self.no - 1 # 원소 갯수 하나 줄이기
